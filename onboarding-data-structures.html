Client 1──1 KYB_Profile        (business clients have one KYB profile)
KYB_Profile 1──N Beneficial_Owner (KYB has multiple UBOs)
Client 1──N Document           (client uploads multiple documents)
Client 1──N AML_Check          (client has multiple AML screenings)
Client 1──N Risk_Assessment    (client has multiple risk evaluations)

Workflow 1──N Workflow_Event    (workflow has event audit trail)
Institution 1──N Client        (institution has many clients)
Institution 1──N Staff         (institution has many staff)
Staff 1──N Notification       (staff receives notifications)

Supabase Migration Example

-- Create Client table with RLS
CREATE TABLE public.client (
  id          uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  institution_id uuid REFERENCES public.institution(id),
  type        text NOT NULL CHECK (type IN ('individual','business')),
  status      text NOT NULL DEFAULT 'pending'
              CHECK (status IN ('pending','in_review','approved','rejected','suspended')),
  email       varchar(255) NOT NULL,
  phone       varchar(50),
  metadata    jsonb DEFAULT '{}',
  created_at  timestamptz DEFAULT now(),
  updated_at  timestamptz DEFAULT now()
);

-- Enable Row Level Security
ALTER TABLE public.client ENABLE ROW LEVEL SECURITY;

-- Institution-scoped access policy
CREATE POLICY "institution_access" ON public.client
  USING (institution_id = auth.jwt()->>'institution_id');

-- Auto-update updated_at
CREATE OR REPLACE FUNCTION update_timestamp()
RETURNS TRIGGER AS $$
BEGIN NEW.updated_at = now(); RETURN NEW; END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER client_updated
  BEFORE UPDATE ON public.client
  FOR EACH ROW EXECUTE FUNCTION update_timestamp();
